def all(params = {})
  params = Utils.symbolize_hash(params)
  url    = articles_path(params.delete(:topic))

  paginate url, :articles, params
end

def articles_path(topic_slug = nil)
  if topic_slug
    "/topics/#{topic_slug}/articles"
  else
    "/articles"
  end
end
